typedef struct node {
    int val;
    struct node* next;
    struct node* pre;
}node, deque;

node* head;
node* rear;
void pushBack(int val) {
    node *new_node;
    new_node = (node*) malloc(sizeof (node));
    if (new_node == nullptr) {
        perror("malloc fail\n");
    }
    new_node -> val = val;
    if (head == nullptr) {
        head = new_node;
        head->pre = nullptr;
        rear = new_node;
    } else {
        rear->next = new_node;
        new_node->pre = rear;
        rear = new_node;
    }
    rear->next = nullptr;
}
void pushFront(int val) {
    node* new_node;
    new_node = (node*) malloc(sizeof(node));
    if (new_node == nullptr) {
        perror("malloc error\n");
    }

    new_node->val = val;
    if (head == nullptr) {
        head = new_node;
        rear = new_node;
        rear -> next = nullptr;
    } else {
        head->pre = new_node;
        new_node->next = head;
        head = new_node;
    }
    head->pre = nullptr;
}
